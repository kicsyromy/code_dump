cmake_minimum_required(VERSION 3.14)

project (rml CXX)

set (CMAKE_CXX_STANDARD 17)
set (CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable (
    ${PROJECT_NAME}
    src/main.cc
)

target_include_directories (
    ${PROJECT_NAME} PRIVATE
    src/include
)

target_link_libraries (
    ${PROJECT_NAME} PRIVATE
    stdc++fs
    pthread
)
